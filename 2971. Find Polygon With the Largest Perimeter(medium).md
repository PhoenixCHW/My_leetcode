class Solution:
    def largestPerimeter(self, nums: List[int]) -> int:
        nums.sort()
        while len(nums) > 2:
            if nums[-1] >= sum(nums[:-1]):
                nums.remove(nums[-1])
            else:
                return sum(nums)
        return -1
