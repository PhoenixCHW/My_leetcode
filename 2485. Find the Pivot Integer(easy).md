```python3
class Solution:
    def pivotInteger(self, n: int) -> int:
        if n == 1:
            return 1
        numIndex = []
        for i in range(1,n+1):
            numIndex.append(i)
        
        for j in range(n):
            if sum(numIndex[0:n-j]) == sum(numIndex[-1:-j-2:-1]):
                return n-j
            elif sum(numIndex[0:n-j]) < sum(numIndex[-1:-j-2:-1]):
                return -1


```
