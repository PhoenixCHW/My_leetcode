# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def middleNode(self, head: Optional[ListNode]) -> Optional[ListNode]:
        cc = head
        size = 1
        while cc.next:
            size += 1
            cc = cc.next
        
        if size == 1:
            return head
        if size == 2:
            return head.next
    
        middle = size // 2 + 1
        ans = head
        count = 1
        while head.next:
            if count == middle:
                ans = head
                break
            head = head.next
            count += 1
        return ans
